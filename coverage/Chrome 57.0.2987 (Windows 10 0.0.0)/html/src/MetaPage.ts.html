	public AddString(value: string): number {
		if (value == undefined)
			value = "";
		let avalue: number = this.stringlist[value];
		if (avalue != undefined)
			return avalue;
		let newindex = this.strings.length;
		this.strings.push(value);
		this.stringlist[value] = newindex;
		return newindex;
	}
